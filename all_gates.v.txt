module all_gates (
    input a,
    input b,
    output out_and,
    output out_or,
    output out_not_a,
    output out_nand,
    output out_nor,
    output out_xor,
    output out_xnor
);

    // Continuous assignment statements mapping directly to boolean logic
    assign out_and   = a & b;
    assign out_or    = a | b;
    assign out_not_a = ~a;       // Inverting only input 'a' for demonstration
    assign out_nand  = ~(a & b);
    assign out_nor   = ~(a | b);
    assign out_xor   = a ^ b;
    assign out_xnor  = ~(a ^ b);

endmodule
